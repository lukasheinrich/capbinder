%matplotlib inline
import matplotlib.pyplot as plt
import requests
import pyhf
import capatlas

wspace = requests.get('https://gist.githubusercontent.com/lukasheinrich/dd680c74daf9655db5b8d44c2b4e229e/raw/13b38786241fe118f344e6af35767b2cfefb7fc7/wspac').json()
pdf, data, selector, nominal, order = capatlas.read_wspace(wspace)

f,(ax) = plt.subplots(1,1, sharey=True, sharex=True)
f.set_size_inches(8,5)

nominal['SigXsecOverSM'] = 1.0
capatlas.plot_lhood(pdf,data, ax = ax, order = order, selector = selector, **nominal)
